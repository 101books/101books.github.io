\def\title#1{%
  \begin{titlepage}%
  \centering%
  \vspace*{60pt}%
  \rule{\textwidth}{8pt}\\%
  \vspace{10pt}%
  {\scshape\HUGE{#1}\\}%
  \vspace{17.4pt}%
  \rule{\textwidth}{8pt}%
  \end{titlepage}%

  \setcounter{page}{0}%
  \pagestyle{plain}%
}


\def\countproblems{%
  \newcounter{pcount}%
  \def\p##1{\stepcounter{pcount}}%
  \problems%
  \thepcount%
}

\def\renderproblems{%
  \def\line#1{\truncate[]{150pt}{#1}\\}%
  \def\halfgoban{%
    \def\line#1{#1\\}%
    \clearpage%
  }
  \def\p##1{%
    \hspace{5pt}%
    \begin{varwidth}{\textwidth}%
      ~\\[4pt]%
      \center%
      \small\input{problems/\jobname/##1.gnos}%
      \vspace{5pt}%
      \center{(##1)}%
    \end{varwidth}%
    \allowbreak%
    \hfil%
  }%
  \newgeometry{margin=40pt, top=50pt, bottom=80pt}%
  \begin{center}%
  \hspace{-5pt}% don't judge
  \problems%
  \end{center}%
  \clearpage%
}
